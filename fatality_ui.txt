-- Get team folder name ("Terrorists" or "Counter-Terrorists")
local function GetTeam(p)
	if not Characters then return nil end
	for _,folder in ipairs(Characters:GetChildren()) do
		if folder:FindFirstChild(p.Name) then return folder.Name end
	end
	return nil
end

local function IsEnemy(p)
	if not p or p == LP or not p.Character then return false end
	-- If no team data available, treat everyone as enemy
	local myTeam  = GetTeam(LP)
	if not myTeam then return true end
	local plrTeam = GetTeam(p)
	if not plrTeam then return true end
	return plrTeam ~= myTeam  -- enemy = different team folder
end

		if type(obj) == "table" then
			for _,v in pairs(obj) do removeObject(v) end
			return

		pcall(function() obj:Remove() end)
	end
	removeObject(o)

	end
	hideObject(o)

-- Chams (team-aware, live update)
local ChamHL = {}

local function ApplyChamsToPlayer(plr)
	-- Remove old highlight
	if ChamHL[plr] then
		pcall(function() ChamHL[plr]:Destroy() end)
		ChamHL[plr] = nil
	end
	if not S.Chams then return end
	if not plr.Character then return end

	-- Skip teammates
	if not IsEnemy(plr) then return end

	task.wait(0.15)
	if not plr.Character then return end

	local hl = Instance.new("Highlight")
	hl.Adornee           = plr.Character
	hl.FillColor         = Color3.fromRGB(220, 40, 40)   -- red = enemy
	hl.OutlineColor      = Color3.fromRGB(255, 255, 255)
	hl.FillTransparency  = math.clamp(S.ChamsFill / 10, 0, 1)
	hl.DepthMode         = Enum.HighlightDepthMode.AlwaysOnTop
	hl.Parent            = plr.Character
	ChamHL[plr]          = hl
end

local function SetChams(plr, on)
	if ChamHL[plr] then
		pcall(function() ChamHL[plr]:Destroy() end)
		ChamHL[plr] = nil
	end
	if on then ApplyChamsToPlayer(plr) end
end

-- Re-apply on respawn
local function HookChamsRespawn(plr)
	plr.CharacterAdded:Connect(function()
		if S.Chams then
			task.wait(0.2)
			ApplyChamsToPlayer(plr)
		end
	end)
end

for _,p in ipairs(Players:GetPlayers()) do
	if p ~= LP then
		HookChamsRespawn(p)
		if S.Chams then ApplyChamsToPlayer(p) end
	end
end
Players.PlayerAdded:Connect(function(p)
	if p ~= LP then
		HookChamsRespawn(p)
	end
end)

-- Refresh chams every 2s (handles team changes)
task.spawn(function()
	while true do
		task.wait(2)
		if S.Chams then
			for _, p in ipairs(Players:GetPlayers()) do
				if p ~= LP then
					-- Remove if now teammate, apply if now enemy
					if IsEnemy(p) then
						if not ChamHL[p] or not ChamHL[p].Parent then
							ApplyChamsToPlayer(p)
						end
					else
						if ChamHL[p] then
							pcall(function() ChamHL[p]:Destroy() end)
							ChamHL[p] = nil
						end
					end
				end
			end
		end
	end
end)

-- Predict future position using current velocity
local function PredictPos(part, predTime)
	local vel = Vector3.zero
	pcall(function() vel = part.AssemblyLinearVelocity end)
	return part.Position + vel * predTime
end

local function IsPartPenetrable(part)
	if not part or not part:IsA("BasePart") then return false end
	if part.Transparency >= 0.5 then return true end
	if not part.CanCollide then return true end
	local mat = part.Material
	local penetrable = {
		[Enum.Material.Glass] = true,
		[Enum.Material.SmoothPlastic] = true,
		[Enum.Material.Plastic] = true,
		[Enum.Material.Neon] = true,
		[Enum.Material.Fabric] = true,
		[Enum.Material.WoodPlanks] = true,
		[Enum.Material.CorrodedMetal] = true,
		[Enum.Material.Wood] = true,
		[Enum.Material.Brick] = true,
		[Enum.Material.Concrete] = true,
		[Enum.Material.Grass] = true,
		[Enum.Material.Slate] = true,
		[Enum.Material.Granite] = true,
		[Enum.Material.Rock] = true,
		[Enum.Material.Metal] = true,
		[Enum.Material.Sand] = true,
		[Enum.Material.Mud] = true,
		[Enum.Material.Salt] = true,
		[Enum.Material.Cobblestone] = true,
		[Enum.Material.CrackedLava] = true,
		[Enum.Material.CorrodedMetal] = true,
		[Enum.Material.Water] = true,
	}
	if penetrable[mat] then return true end
	local lname = part.Name:lower()
	if lname:find("glass") or lname:find("window") or lname:find("wall") or lname:find("door") then
		return true
	end
	return false
end

local function IsVisible(part, allowPenetration)
	if not part then return false end
	local myChar = LP.Character
	local targetRoot = part.Parent or part
	local exclude = {Camera}
	if myChar then exclude[#exclude+1] = myChar end
	exclude[#exclude+1] = targetRoot
	local params = RaycastParams.new()
	params.FilterType = Enum.RaycastFilterType.Exclude
	params.FilterDescendantsInstances = exclude
	local origin = Camera.CFrame.Position
	local dir    = part.Position - origin
	local result = workspace:Raycast(origin, dir, params)
	if not result then return true end  -- nothing in way = visible
	if result.Instance and result.Instance:IsDescendantOf(targetRoot) then return true end

	local blocker = result.Instance
	if not allowPenetration then
		return false, blocker
	end
	if not IsPartPenetrable(blocker) then
		return false, blocker
	end

	local throughExclude = {unpack(exclude)}
	throughExclude[#throughExclude+1] = blocker
	local throughParams = RaycastParams.new()
	throughParams.FilterType = Enum.RaycastFilterType.Exclude
	throughParams.FilterDescendantsInstances = throughExclude
	local startPos = result.Position + dir.Unit * 0.1
	local remaining = part.Position - startPos
	if remaining.Magnitude <= 0 then
		return true, blocker
	end
	local secondResult = workspace:Raycast(startPos, remaining, throughParams)
	if not secondResult then return true, blocker end
	return secondResult.Instance and secondResult.Instance:IsDescendantOf(part.Parent), blocker
end

-- GetClosest: full rage logic with TargetSel + MinDamage + WallCheck
local function GetClosest(fov, headOnly, fromMouse, wallCheck, prediction)
	local candidates = {}
	local ref = fromMouse and UIS:GetMouseLocation() or GetCenter()
	local scaledFov = fov * math.clamp(S.PointScale, 1, 100) / 100
	for _, p in ipairs(Players:GetPlayers()) do
		if IsEnemy(p) then
			local char = GetCharModel(p)
			local hum  = char and char:FindFirstChildOfClass("Humanoid")
			if char and hum and hum.Health > 0 then
				if not (S.MinDamage > 0 and hum.Health < S.MinDamage) then
					local part = FindAimPart(char, headOnly or (S.HitboxMode == "Head"))
					if part then
						if not ((wallCheck or S.WallCheck) and not IsVisible(part, S.PenetrationCheck)) then
							local targetPos = prediction and PredictPos(part, 0.07) or part.Position
							local sp, onScreen = Camera:WorldToViewportPoint(targetPos)
							if onScreen and sp.Z > 0 then
								local screenDist = (Vector2.new(sp.X, sp.Y) - ref).Magnitude
								if screenDist <= scaledFov then
									local worldDist = (part.Position - Camera.CFrame.Position).Magnitude
									-- if prediction requested, compute predTime based on distance and multiplier
									local finalPos = targetPos
									if prediction then
										local predTime = math.clamp((worldDist/1000) * (S.RagePredictMultiplier or 1), 0.02, 0.25)
										finalPos = PredictPos(part, predTime)
									end
									table.insert(candidates, {
										part=part, pos=finalPos, plr=p,
										screenDist=screenDist,
										health=hum.Health,
										worldDist=worldDist,
									})
								end
							end
						end
					end
				end
			end
		end
	end
	if #candidates == 0 then return nil end
	-- Sort by TargetSel
	local sel = S.TargetSel or "FOV"
	table.sort(candidates, function(a, b)
		if sel == "Health"   then return a.health   < b.health   end
		if sel == "Distance" then return a.worldDist < b.worldDist end
		if sel == "Threat" then
			-- threat = weighted score: prefer low health and close distance and within FOV
			local aScore = a.screenDist * 0.5 + a.worldDist * 0.35 + a.health * 0.15
			local bScore = b.screenDist * 0.5 + b.worldDist * 0.35 + b.health * 0.15
			return aScore < bScore
		end
		return a.screenDist < b.screenDist -- FOV (default)
	end)
	return candidates[1]
end

local function FindWeaponData(name, wf)

local function SetNumericValue(root, key, value)
	if not root or not key then return false end
	local success = false
	pcall(function()
		if type(root) == "table" then
			local prop = rawget(root, key)
			if type(prop) == "number" then
				root[key] = value
				success = true
				return
			end
			if type(prop) == "table" and prop.Value and type(prop.Value) == "number" then
				prop.Value = value
				success = true
				return
			end
		else
			local child = root:FindFirstChild(key)
			if child and type(child.Value) == "number" then
				child.Value = value
				success = true
				return
			end
			local prop = root[key]
			if type(prop) == "number" then
				root[key] = value
				success = true
				return
			end
		end
	end)
	return success
end

local function PatchGunValuesDescendants(root, keys, value)
	if not root or type(root) ~= "Instance" then return end
	for _, desc in ipairs(root:GetDescendants()) do
		if desc:IsA("NumberValue") or desc:IsA("IntValue") then
			for _, key in ipairs(keys) do
				if desc.Name == key then
					pcall(function() desc.Value = value end)
				end
			end
		end
	end
end

local function PatchAllWeapons(keys, value)
	local roots = {}
	local mainWeapons = GetWeaponsFolder()
	if mainWeapons then roots[#roots+1] = mainWeapons end
	if workspace:FindFirstChild("demoModels") and workspace.demoModels:FindFirstChild("weapons") then
		roots[#roots+1] = workspace.demoModels.weapons
	end
	for _, root in ipairs(roots) do
		PatchGunValuesDescendants(root, keys, value)
	end
end

local function PatchEnvValues(env, keys, value)
	if not env then return end
	pcall(function()
		for _, key in ipairs(keys) do
			if env[key] ~= nil then
				local ok, valType = pcall(function() return typeof(env[key]) end)
				if ok then
					if valType == "number" then
						env[key] = value
					elseif valType == "Vector2" and type(value) == "number" and value == 0 then
						env[key] = Vector2.new(0,0)
					elseif valType == "Vector3" and type(value) == "number" and value == 0 then
						env[key] = Vector3.new(0,0,0)
					end
				end
			end
		end
	end)
end

local function PatchGunValues(root, keys, value)
	if not root then return end
	local patched = false
	for _, key in ipairs(keys) do
		if SetNumericValue(root, key, value) then
			patched = true
		end
	end
	if typeof(root) == "Instance" then
		PatchGunValuesDescendants(root, keys, value)
	end
	return patched
end

local Lighting = game:GetService("Lighting")
local WorldVisualsState = {orig={}}

local function SaveWorldVisuals()
	if WorldVisualsState.orig.Saved then return end
	local l = Lighting
	WorldVisualsState.orig = {
		Brightness = l.Brightness,
		Ambient = l.Ambient,
		OutdoorAmbient = l.OutdoorAmbient,
		FogStart = l.FogStart,
		FogEnd = l.FogEnd,
		FogColor = l.FogColor,
		ColorShiftBottom = l.ColorShift_Bottom,
		ColorShiftTop = l.ColorShift_Top,
		Effects = {}
	}
	for _, child in ipairs(l:GetChildren()) do
		if child:IsA("BloomEffect") or child:IsA("ColorCorrectionEffect") or child:IsA("SunRaysEffect") then
			WorldVisualsState.orig.Effects[child.ClassName .. child.Name] = child:Clone()
		end
	end
	WorldVisualsState.orig.Saved = true
end

local function RestoreWorldVisuals()
	if not WorldVisualsState.orig.Saved then return end
	local l = Lighting
	l.Brightness = WorldVisualsState.orig.Brightness
	l.Ambient = WorldVisualsState.orig.Ambient
	l.OutdoorAmbient = WorldVisualsState.orig.OutdoorAmbient
	l.FogStart = WorldVisualsState.orig.FogStart
	l.FogEnd = WorldVisualsState.orig.FogEnd
	l.FogColor = WorldVisualsState.orig.FogColor
	l.ColorShift_Bottom = WorldVisualsState.orig.ColorShiftBottom
	l.ColorShift_Top = WorldVisualsState.orig.ColorShiftTop
	for _, child in ipairs(l:GetChildren()) do
		if child:IsA("BloomEffect") or child:IsA("ColorCorrectionEffect") or child:IsA("SunRaysEffect") then
			child:Destroy()
		end
	end
	for _, clone in pairs(WorldVisualsState.orig.Effects) do
		local restored = clone:Clone()

-- Watch enemies for hit detection
local LastFire=0; local LastTrigger=0; local LastGraphUpdate=0
local function IsTargetLikelyFromLocal(char)
	if not char or not Camera then return false end
	local part = char:FindFirstChild("Head") or char:FindFirstChild("HumanoidRootPart")
	if not part then return false end
	if not IsVisible(part) then return false end
	local sp, onScreen = Camera:WorldToViewportPoint(part.Position)
	if not onScreen or sp.Z <= 0 then return false end
	local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
	return (Vector2.new(sp.X, sp.Y) - center).Magnitude <= 90
end

local function WatchPlayer(p)
	if p == LP then return end
	local function onChar(char)
		local hum = char:WaitForChild("Humanoid",5)
		if not hum then return end
		local last = hum.Health
		hum.HealthChanged:Connect(function(hp)
			if not IsEnemy(p) then last=hp; return end
			if hp < last then
				local dmg = math.floor(last-hp)
				local now = tick()
				if now - LastFire < 0.5 and IsTargetLikelyFromLocal(char) then
					local isHS = (dmg >= 90) or (hum.MaxHealth>0 and hp<=0)
					local isKill = hp <= 0
					if S.HitNotifEnabled then
						ShowHitNotif(p.Name, dmg, isHS, isKill)
					end
					FlashHit(isHS)
					PlayHitSound(isHS)
				end
			end
			last = hp
		end)
	end
	p.CharacterAdded:Connect(onChar)
	if p.Character then task.spawn(onChar, p.Character) end
end
for _, p in ipairs(Players:GetPlayers()) do WatchPlayer(p) end
Players.PlayerAdded:Connect(WatchPlayer)

-- Simple tracer: draws a short-lived screen line between two world points
local function GetFireOrigin(tool)
	if tool and tool:IsA("Instance") then
		local muzzle = tool:FindFirstChild("Muzzle") or tool:FindFirstChild("muzzle")
			or tool:FindFirstChild("Barrel") or tool:FindFirstChild("Handle")
		if muzzle and muzzle:IsA("BasePart") then
			return muzzle.Position
		end
		if tool.PrimaryPart and tool.PrimaryPart:IsA("BasePart") then
			return tool.PrimaryPart.Position
		end
	end
	if Camera then return Camera.CFrame.Position end
	return workspace.CurrentCamera and workspace.CurrentCamera.CFrame.Position
end

local function GetFireDirection(tool)
	if tool and tool:IsA("Instance") then
		local muzzle = tool:FindFirstChild("Muzzle") or tool:FindFirstChild("muzzle")
			or tool:FindFirstChild("Barrel") or tool:FindFirstChild("Handle")
		if muzzle and muzzle:IsA("BasePart") then
			return muzzle.CFrame.LookVector
		end
		if tool.PrimaryPart and tool.PrimaryPart:IsA("BasePart") then
			return tool.PrimaryPart.CFrame.LookVector
		end
	end
	if Camera then return Camera.CFrame.LookVector end
	return workspace.CurrentCamera and workspace.CurrentCamera.CFrame.LookVector or Vector3.new(0,0,-1)
end

local function SpawnTracer(fromWorld, toWorld, color)
	pcall(function()
		if not Camera then Camera = workspace.CurrentCamera end
		local fs = Camera:WorldToViewportPoint(fromWorld)
		local ts = Camera:WorldToViewportPoint(toWorld)
		if fs.Z <= 0 or ts.Z <= 0 then return end
		local from2 = Vector2.new(fs.X, fs.Y)
		local to2 = Vector2.new(ts.X, ts.Y)
		local segs = {}
		local mid = (from2 + to2) / 2
		for i=1,4 do
			local t = i / 4
			local offset = Vector2.new((math.random()-0.5)*24, (math.random()-0.5)*24) * (1 - t)
			local a = from2:Lerp(mid, t * 0.9) + offset
			local b = from2:Lerp(to2, t * 0.9) + offset * 0.3
			local ln = Drawing.new("Line")
			ln.From = a
			ln.To = b
			ln.Color = color or Color3.fromRGB(255,200,80)
			ln.Thickness = 2
			ln.Transparency = 1
			ln.Visible = true
			segs[#segs+1] = ln
		end
		local spark = Drawing.new("Circle")
		spark.Position = to2
		spark.Radius = 4
		spark.Filled = true
		spark.Color = color or Color3.fromRGB(255,200,80)
		spark.Transparency = 0.8
		spark.Visible = true
		task.spawn(function()
			local life = 0.14
			local t0 = tick()
			while tick()-t0 < life do
				local p = 1 - (tick()-t0)/life
				for _,ln in ipairs(segs) do
					pcall(function() ln.Transparency = math.clamp(p,0,1) end)
				end
				pcall(function() spark.Transparency = math.clamp(p*0.8,0,1) end)
				task.wait()
			end
			for _,ln in ipairs(segs) do pcall(function() ln.Visible=false; ln:Remove() end) end
			pcall(function() spark.Visible=false; spark:Remove() end)
		end)
	end)
end

local function IsLocalStationary()
	local char = LP.Character
	if not char then return false end
	local root = char:FindFirstChild("HumanoidRootPart")
	local hum  = char:FindFirstChildOfClass("Humanoid")
	if not root or not hum then return false end
	local vel = root.AssemblyLinearVelocity
	local moving = Vector3.new(vel.X, 0, vel.Z).Magnitude > 1.5
	local airborne = hum.FloorMaterial == Enum.Material.Air
	return not moving and not airborne
end

	if not targetPos then return end
	local sp = Camera:WorldToViewportPoint(targetPos)
	if sp.Z <= 0 then return end
	local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
	local delta = Vector2.new(sp.X, sp.Y) - center
	if math.abs(delta.X) < 1 and math.abs(delta.Y) < 1 then return end
	if smooth and smooth > 0 and type(_mmr) == "function" then
		pcall(_mmr, delta.X / smooth, delta.Y / smooth)
	else
		pcall(_mmr, delta.X, delta.Y)
	end
end

local function CanStandshotFire()
	return IsLocalStationary()
end

local function TrackLocalFire()
	local function hookTool(tool)
		if not tool or not tool:IsA("Tool") then return end
		tool.Activated:Connect(function()
			LastFire = tick()
			pcall(function()
				if not Camera then Camera = workspace.CurrentCamera end
				local origin = GetFireOrigin(tool)
				local dir = GetFireDirection(tool) * 2000
				local params = RaycastParams.new()
				params.FilterType = Enum.RaycastFilterType.Exclude
				params.FilterDescendantsInstances = {LP.Character}
				local res = workspace:Raycast(origin, dir, params)
				local toPos = res and res.Position or (origin + dir)
				if S.Tracers then SpawnTracer(origin, toPos, Color3.fromRGB(255,200,80)) end
				if S.FireSoundEnabled and _fireSnd then
					pcall(function() _fireSnd:Play() end)
				end
				if res and res.Instance then
					local isHead = (res.Instance.Name == "Head")
					pcall(function() PlayHitSound(isHead) end)
				end
			end)
		end)
	end

	local function onCharacter(char)
		if not char then return end
		for _, child in ipairs(char:GetChildren()) do
			if child:IsA("Tool") then hookTool(child) end
		end
		char.ChildAdded:Connect(function(child)
			if child:IsA("Tool") then hookTool(child) end
		end)
	end

	if LP.Character then onCharacter(LP.Character) end
	LP.CharacterAdded:Connect(onCharacter)
	UIS.InputBegan:Connect(function(input, gpe)
		if gpe then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			LastFire = tick()
		end
	end)
end
TrackLocalFire()

local function Standshot(target)
	if not target or not S.Triggerbot then return false end
	if S.WallCheck and not IsVisible(target, S.PenetrationCheck) then return false end
	if CanStandshotFire() then
		FireWeapon()
		return true
	end
	AimAtTarget(target.Position, S.RageSmooth)
	return false
end

local function GetCurrentSpread()
	if S.NoSpread then return 0.01 end  -- Very small spread to avoid detection
	local char = LP.Character
	if not char then return 1 end
	local hrp = char:FindFirstChild("HumanoidRootPart")
	local hum = char:FindFirstChildOfClass("Humanoid")
	if not hrp or not hum then return 1 end
	-- Base spread
	local spread = 0.1
	-- Movement spread
	local vel = hrp.AssemblyLinearVelocity
	local speed = Vector3.new(vel.X, 0, vel.Z).Magnitude
	spread = spread + (speed / 50) * 0.5  -- Increase with speed
	-- Airborne spread
	if hum.FloorMaterial == Enum.Material.Air then
		spread = spread + 0.3
	end
	-- Aiming reduces spread (assume ADS if camera zoomed or something, but hard to detect)
	-- For simplicity, assume not aiming increases spread
	spread = spread + 0.2  -- Base non-aiming spread
	return math.clamp(spread, 0.1, 2)
end

local function UpdateSpeedGraph()
	if not S.SpeedGraph then
		if SG.UI then SG.UI.Visible = false end
		return
	end
	local now = tick()
	if now - LastGraphUpdate < 0.05 then return end
	LastGraphUpdate = now
	InitSpeedGraph()
	SG.UI.Visible = true
	-- Calc speed from position delta
	local char = LP.Character
	local hrp  = char and char:FindFirstChild("HumanoidRootPart")
	local spd  = 0
	if hrp then
		local gp = hrp.Position
		if SG.LastPos and SG.LastTime then
			local dtt = now - SG.LastTime
			if dtt > 0 then
				spd = Vector3.new(gp.X-SG.LastPos.X, 0, gp.Z-SG.LastPos.Z).Magnitude / dtt
			end
		end
		SG.LastPos  = gp
		SG.LastTime = now
	end
	SG.Smoothed = SG.Smoothed + (spd - SG.Smoothed) * 0.3
	SG.History[#SG.History+1] = math.min(SG.Smoothed, SG_MAXSPD)
	if #SG.History > SG_LINE_COUNT then table.remove(SG.History, 1) end
	SG.Label.Text = string.format("%.1f", math.floor(SG.Smoothed*10+0.5)/10)
	-- Peak tracking
	if S.SpeedGraphPeak then
		SG.PeakHistory[#SG.PeakHistory+1] = {time=now, speed=SG.Smoothed}
		while #SG.PeakHistory>0 and (now-SG.PeakHistory[1].time)>5 do
			table.remove(SG.PeakHistory,1)
		end
		local peak=0
		for _,e in ipairs(SG.PeakHistory) do if e.speed>peak then peak=e.speed end end
		SG.PeakLabel.Text = string.format("Peak: %.1f", math.floor(peak*10+0.5)/10)
		SG.PeakLabel.TextColor3 = peak>=30 and Color3.fromRGB(255,50,50)
			or peak>=20 and Color3.fromRGB(255,255,50)
			or Color3.new(1,1,1)
		SG.PeakLabel.Visible = true
		-- Avg
		table.insert(SG.AvgHistory, SG.Smoothed)
		if #SG.AvgHistory > 100 then table.remove(SG.AvgHistory, 1) end
		local sum = 0
		for _,v in ipairs(SG.AvgHistory) do sum = sum + v end
		local avg = sum / #SG.AvgHistory
		SG.AvgLabel.Text = string.format("Avg: %.1f", math.floor(avg*10+0.5)/10)
		SG.AvgLabel.Visible = true
	else
		SG.PeakLabel.Visible = false
		SG.AvgLabel.Visible = false
	end
	-- Draw lines
	local w = 300  -- fixed width
	local h = 120   -- fixed height
	local step = w / SG_LINE_COUNT
	for i = 1, SG_LINE_COUNT-1 do
		local gl = SG.Lines[i]
		if gl and i < #SG.History then
			local h1 = (SG.History[i]   or 0) / SG_MAXSPD * h
			local h2 = (SG.History[i+1] or 0) / SG_MAXSPD * h
			local x1,y1 = (i-1)*step, h-h1
			local x2,y2 = i*step,     h-h2
			local gv = Vector2.new(x2-x1, y2-y1)
			local ct = Vector2.new((x1+x2)/2, (y1+y2)/2)
			gl.Position = UDim2.new(0,ct.X,0,ct.Y)
			gl.Size     = UDim2.new(0,gv.Magnitude,0,2)
			gl.Rotation = math.deg(math.atan2(gv.Y,gv.X))
			gl.BackgroundColor3 = SG_COLOR
			gl.Visible  = true
		elseif gl then gl.Visible=false end
	end
end

local function UpdateCrosshair()

	local cx = Camera.ViewportSize.X/2
	local cy = Camera.ViewportSize.Y/2
	local sz = S.CrosshairSize

	local th = S.CrosshairThick
	local style = S.CrosshairStyle

	-- Hide all first
	for i=1,4 do CH[i].Visible=false end
	CH.dot.Visible=false; CH.outline.Visible=false

	if not S.CrosshairEnabled then return end

	-- Dynamic gap: expand while moving
	local dyn = gp
	local char = LP.Character
	if char then
		local hrp = char:FindFirstChild("HumanoidRootPart")
		if hrp then
			local spd = Vector2.new(hrp.AssemblyLinearVelocity.X, hrp.AssemblyLinearVelocity.Z).Magnitude
			dyn = gp + math.min(spd * 0.15, 8)
		end
	end

	if style == "Classic" or style == "Cross" then
		-- Right
		CH[1].From=Vector2.new(cx+dyn,cy); CH[1].To=Vector2.new(cx+dyn+sz,cy)
		CH[1].Thickness=th; CH[1].Color=col; CH[1].Visible=true

		CH[2].From=Vector2.new(cx-dyn,cy); CH[2].To=Vector2.new(cx-dyn-sz,cy)
		CH[2].Thickness=th; CH[2].Color=col; CH[2].Visible=true

		CH[3].Thickness=th; CH[3].Color=col; CH[3].Visible=true

			CH[4].From=Vector2.new(cx,cy-dyn); CH[4].To=Vector2.new(cx,cy-dyn-sz)
			CH[4].Thickness=th; CH[4].Color=col; CH[4].Visible=true
		end
		if style=="Classic" then
			CH.dot.Position=Vector2.new(cx,cy); CH.dot.Radius=th
			CH.dot.Color=col; CH.dot.Visible=true
		end
	elseif style == "Dot" then
		CH.dot.Position=Vector2.new(cx,cy); CH.dot.Radius=sz/4
		CH.dot.Color=col; CH.dot.Visible=true
	elseif style == "Circle" then
		CH.outline.Position=Vector2.new(cx,cy); CH.outline.Radius=sz+dyn
		CH.outline.Color=col; CH.outline.Thickness=th; CH.outline.Visible=true
	end

-- ── Character Skin Changer ───────────────────────────
S.CharSkinEnabled = false
S.CharSkinUserId  = ""

local function ApplyCharSkin(userId)
	local uid = tonumber(userId)
	if not uid or uid <= 0 then return end
	local ok, model = pcall(function()
		return Players:CreateHumanoidModelFromUserId(uid)
	end)
	if not ok or not model then
		warn("[Fatality] CharSkin failed"); return
	end
	local char = LP.Character; if not char then return end
	-- Copy clothing
	for _,cls in ipairs({"Shirt","Pants"}) do
		local old = char:FindFirstChildOfClass(cls)
		if old then old:Destroy() end
		local src = model:FindFirstChildOfClass(cls)
		if src then src:Clone().Parent = char end
	end
	-- Copy body colors
	local srcBC = model:FindFirstChildOfClass("BodyColors")
	local dstBC = char:FindFirstChildOfClass("BodyColors")
	if srcBC and dstBC then
		for _,prop in ipairs({"HeadColor3","TorsoColor3","LeftArmColor3","RightArmColor3","LeftLegColor3","RightLegColor3"}) do
			pcall(function() dstBC[prop] = srcBC[prop] end)
		end
	end
	-- Copy face
	local head    = char:FindFirstChild("Head")
	local srcHead = model:FindFirstChild("Head")
	if head and srcHead then
		for _,d in ipairs(head:GetChildren()) do if d:IsA("Decal") then d:Destroy() end end
		for _,d in ipairs(srcHead:GetChildren()) do if d:IsA("Decal") then d:Clone().Parent=head end end
	end
	-- Copy accessories
	for _,acc in ipairs(char:GetChildren()) do if acc:IsA("Accessory") then acc:Destroy() end end
	for _,acc in ipairs(model:GetChildren()) do
		if acc:IsA("Accessory") then acc:Clone().Parent = char end
	end
	model:Destroy()
	warn("[Fatality] Skin applied: "..tostring(uid))
end

	S.CharSkinUserId = tostring(id)
	if S.CharSkinEnabled then ApplyCharSkin(id) end
end

local _gsAngle=0
AC(RunService.RenderStepped:Connect(function(dt)
	-- Third Person
	pcall(SetThirdPerson, S.ThirdPerson)

	pcall(UpdateWorldVisuals)

	SpreadCircle.Radius   = GetCurrentSpread() * 100  -- Scale for visibility
	SpreadCircle.Visible  = S.ShowSpreadCircle

	-- Speed Graph update
	pcall(UpdateSpeedGraph)

	-- Bhop
	if S.BunnyHop then
		local char=LP.Character
		if char then
			local hum=char:FindFirstChildOfClass("Humanoid")
			local root=char:FindFirstChild("HumanoidRootPart")
			if hum and root and UIS:IsKeyDown(Enum.KeyCode.Space) then
				if hum.FloorMaterial~=Enum.Material.Air then
					hum:ChangeState(Enum.HumanoidStateType.Jumping)
				end
				-- Strafe boost
				local vel=root.AssemblyLinearVelocity
				local camLook=Camera.CFrame.LookVector
				local moveDir=Vector3.new(camLook.X,0,camLook.Z).Unit
				if vel.Magnitude>0.5 and hum.FloorMaterial==Enum.Material.Air then
					local speed=math.min(vel.Magnitude+0.4, hum.WalkSpeed*2.2)
					local horiz=Vector3.new(vel.X,0,vel.Z)
					if horiz.Magnitude>0 then
						root.AssemblyLinearVelocity=Vector3.new(
							horiz.Unit.X*speed, vel.Y, horiz.Unit.Z*speed)
					end
				end
			end
		end
	end

	-- Legit Aimbot (bankroll method — p.Character direct, target lock)
	do
		local SC = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
		local akd = S.LegitAim and UIS:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)
		if akd then
			local bt, bd = nil, S.LegitFOV
			for _, p in ipairs(Players:GetPlayers()) do
				if IsEnemy(p) then
					local char = p.Character  -- bankroll: use p.Character directly
					if char then
						local hum = char:FindFirstChildOfClass("Humanoid")
						if hum and hum.Health > 0 then
							local partName = S.HeadshotOnly and "Head" or "HumanoidRootPart"
							local pt = char:FindFirstChild(partName)
							if pt then
								local sp, ao = Camera:WorldToViewportPoint(pt.Position)
								if ao and IsVisible(pt, S.PenetrationCheck) then
									local ad = (Vector2.new(sp.X, sp.Y) - SC).Magnitude
									if ad < bd then bt = Vector2.new(sp.X, sp.Y); bd = ad end
								end
							end
						end
					end
				end
			end
			if bt then
				local sm = math.max(1, S.LegitSmooth)
				pcall(_mmr, (bt.X - SC.X) / sm, (bt.Y - SC.Y) / sm)
			end
		end
	end

	-- Auto Fire (rage)
	if S.AutoFire or S.ForceShoot then
		local t = GetClosest(S.SilentAimFOV, S.HeadshotOnly, true, false, S.RagePredict)
		if t then
			if S.RageEnabled and not S.SilentAim and not S.NoMoveCamera then
				AimAtTarget(t.pos, S.RageSmooth)
			end
			local canHit, blocker = IsVisible(t.part, S.PenetrationCheck)
			if not S.WallCheck or canHit then
				FireWeapon()
			end
		end
	end

	-- Anti-Aim
	if S.AntiAim then
		local char=LP.Character
		if char then
			local root=char:FindFirstChild("HumanoidRootPart")
			local hum=char:FindFirstChildOfClass("Humanoid")
			if root and hum then
				hum.AutoRotate=false
				if S.AntiAimMode=="Spin" then
					_gsAngle=_gsAngle+math.rad(S.SpinSpeed)
					root.CFrame=root.CFrame*CFrame.Angles(0,_gsAngle,0)
				elseif S.AntiAimMode=="Backward" then
					local a=math.atan2(Camera.CFrame.LookVector.X,Camera.CFrame.LookVector.Z)
					root.CFrame=CFrame.new(root.Position)*CFrame.Angles(0,a+math.pi,0)
				end
			end
		end
	elseif LP.Character then
		local hum=LP.Character:FindFirstChildOfClass("Humanoid")
		if hum then hum.AutoRotate=true end
	end

	-- Triggerbot (raycast, BloxStrike character detection)
	if S.Triggerbot then
		local now = tick()
		if now - LastTrigger > S.TriggerDelay/1000 then
			local ro = Camera.CFrame.Position
			local rd = Camera.CFrame.LookVector * 1000
			local tp = RaycastParams.new()
			tp.FilterType = Enum.RaycastFilterType.Exclude
			local mc = LP.Character
			tp.FilterDescendantsInstances = mc and {mc, Camera} or {Camera}
			local rr = workspace:Raycast(ro, rd, tp)
			if rr and rr.Instance then
				-- Walk up to find model
				local model = rr.Instance
				while model and not model:FindFirstChildOfClass("Humanoid") do
					model = model.Parent
				end
				local hum = model and model:FindFirstChildOfClass("Humanoid")
				if hum and hum.Health > 0 then
					-- Check if this model belongs to an enemy player
					-- Check by model name OR GetPlayerFromCharacter
					local hpp = Players:GetPlayerFromCharacter(model)
					if not hpp then
						-- BloxStrike fallback: match by name
						for _, p in ipairs(Players:GetPlayers()) do
							if p.Name == model.Name and p ~= LP then hpp = p; break end
						end
					end
					if hpp and hpp ~= LP and IsEnemy(hpp) then
						local targetPart = model:FindFirstChild("Head") or model:FindFirstChild("HumanoidRootPart")
						if targetPart then
							-- Check if target is in FOV
							local sp, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
							local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
							local dist = (Vector2.new(sp.X, sp.Y) - center).Magnitude
							local fovRadius = math.tan(math.rad(S.SilentAimFOV or 180) / 2) * (sp.Z or 1000)
							if onScreen and sp.Z > 0 and dist < fovRadius then
								if S.Standshot then
									if Standshot(targetPart) then
										LastTrigger = now
									end
								else
									local canHit, blocker = IsVisible(targetPart, S.PenetrationCheck)
									if not S.WallCheck or canHit then
										if not S.AdaptiveSpread or GetCurrentSpread() < S.SpreadThreshold then
											LastTrigger = now
											pcall(FireWeapon)
										end
									end
								end
							end
						end
					end
				end
			end
		end
	end

	-- ESP
	Camera = workspace.CurrentCamera or Camera
	local espOn=S.BoxESP or S.NameESP or S.HealthBar or S.DistESP or S.WeaponESP or S.Tracers
	local activePenParts = {}
	for _,p in ipairs(Players:GetPlayers()) do
		local o=ESPObjects[p]
		if o then
			if espOn and IsEnemy(p) and IsAlive(p) then
				local char=p.Character
				if char then
					local hrp=char:FindFirstChild("HumanoidRootPart")
					local head=char:FindFirstChild("Head")
					local hum=char:FindFirstChildOfClass("Humanoid")
					if hrp and head and hum then
						local hSP,hOn = Camera:WorldToViewportPoint(head.Position)
						local fSP,fOn = Camera:WorldToViewportPoint(hrp.Position-Vector3.new(0,3,0))
						local visible = (hOn or fOn) and (hSP.Z > 0 or fSP.Z > 0)
						local x1 = hSP.X; local x2 = fSP.X
						local y1 = hSP.Y; local y2 = fSP.Y
						local minX = math.min(x1, x2)
						local maxX = math.max(x1, x2)
						local minY = math.min(y1, y2)
						local maxY = math.max(y1, y2)
						if visible and minX <= Camera.ViewportSize.X and maxX >= 0
							and minY <= Camera.ViewportSize.Y and maxY >= 0 then
							local w = math.max(38, maxX - minX + 40)
							local h = math.max(40, maxY - minY + 36)
							local x = minX - (w - (maxX - minX)) / 2
							local y = minY - (h - (maxY - minY)) / 2
							local dist=(hrp.Position-Camera.CFrame.Position).Magnitude
							local visibleAim = IsVisible(head, false)
							local color = Color3.fromRGB(220,40,40)
							if not visibleAim then
								color = Color3.fromRGB(155,155,155)
							end
							if S.ESPColorMode == "Rainbow" then
								color = Color3.fromHSV((tick()*0.2) % 1, 1, 1)
							elseif S.ESPColorMode == "Distance" then
								local ratio = math.clamp(dist/120,0,1)

							end
							if S.PenetrationCheck then
								local _, blocker = IsVisible(head, S.PenetrationCheck)
								if blocker then
									local canPen = IsPartPenetrable(blocker)

							end
							local boxFill = S.ESPStyle == "Filled"

							-- Box / filled

							o.box.Color=color; o.box.Filled=boxFill; o.box.Transparency = boxFill and 0.65 or 0

	spawned=true

		end

-- Auto-bind helper: map control names to keys in `S` (or getgenv().FatalityState) when user interacts.
local function bindToStateByName(nm, val)
	if not nm then return end
	local name = tostring(nm)
	local key = name:gsub("%W","")
	local candidates = {
		key,
		key:sub(1,1):lower()..key:sub(2),
		key:sub(1,1):upper()..key:sub(2),
	}
	for _,k in ipairs(candidates) do
		if S[k] ~= nil then S[k] = val; if getgenv then getgenv().FatalityState = S end; return end
	end
	-- fallback: set on global state
	if getgenv then getgenv().FatalityState = getgenv().FatalityState or S; getgenv().FatalityState[key] = val end
end

local function attachAutoCallbackToSection(section)
	if not section or type(section) ~= 'table' then return end
	local function wrap(orig)
		return function(self, t)
			t = t or {}
			local nm = t.Name or t.name or ''
			if not t.Callback then
				t.Callback = (function(nm2)
					return function(v) bindToStateByName(nm2, v) end
				end)(nm)
			end
			return orig(self, t)
		end
	end
	if section.AddToggle and not section._fatality_wrapped_toggle then section.AddToggle = wrap(section.AddToggle); section._fatality_wrapped_toggle = true end
	if section.AddSlider and not section._fatality_wrapped_slider then section.AddSlider = wrap(section.AddSlider); section._fatality_wrapped_slider = true end
	if section.AddDropdown and not section._fatality_wrapped_dropdown then section.AddDropdown = wrap(section.AddDropdown); section._fatality_wrapped_dropdown = true end
	if section.AddButton and not section._fatality_wrapped_button then section.AddButton = wrap(section.AddButton); section._fatality_wrapped_button = true end
	if section.AddKeybind and not section._fatality_wrapped_keybind then section.AddKeybind = wrap(section.AddKeybind); section._fatality_wrapped_keybind = true end
end

-- Try to attach to any existing menus/sections inside Window (best-effort)
pcall(function()
	for _,menu in pairs(Window) do
		if type(menu) == 'table' then
			for _,sec in pairs(menu) do
				attachAutoCallbackToSection(sec)
			end
			-- wrap AddSection on menu so future sections get wrapped
			if menu.AddSection and not menu._fatality_wrapped_addsection then
				local origAS = menu.AddSection
				menu.AddSection = function(self, arg)
					local sec = origAS(self, arg)
					attachAutoCallbackToSection(sec)
					return sec
				end
				menu._fatality_wrapped_addsection = true
			end
		end
	end
	-- wrap Window.AddMenu so menus created later will have their sections wrapped
	if Window.AddMenu and not Window._fatality_wrapped_addmenu then
		local origAM = Window.AddMenu
		Window.AddMenu = function(self, arg)
			local menu = origAM(self, arg)
			pcall(function()
				for _,sec in pairs(menu) do attachAutoCallbackToSection(sec) end
				if menu.AddSection and not menu._fatality_wrapped_addsection then
					local origAS2 = menu.AddSection
					menu.AddSection = function(self2, arg2)
						local sec2 = origAS2(self2, arg2)
						attachAutoCallbackToSection(sec2)
						return sec2
					end
					menu._fatality_wrapped_addsection = true
				end
			end)
			return menu
		end
		Window._fatality_wrapped_addmenu = true
	end
end)
